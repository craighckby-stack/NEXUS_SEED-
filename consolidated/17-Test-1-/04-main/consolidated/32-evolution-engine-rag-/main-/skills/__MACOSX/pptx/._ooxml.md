/**
 * Removes macOS-specific metadata from a given file path.
 *
 * @param {string} filePath - The file path to process.
 * @returns {string} The file path with macOS metadata removed.
 */
function removeMacOSMetadata(filePath) {
  return filePath.split('/').filter(directory => !directory.startsWith('__MACOSX')).join('/');
}

// Example usage:
const originalFilePath = 'skills/__MACOSX/pptx/._ooxml.md';
const cleanedFilePath = removeMacOSMetadata(originalFilePath);
console.log(cleanedFilePath); // Output: skills/pptx/._ooxml.md
