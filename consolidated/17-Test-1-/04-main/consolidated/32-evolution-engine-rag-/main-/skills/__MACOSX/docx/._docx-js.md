// Define a function to parse Mac OS X metadata
/**
 * Extracts and returns relevant quarantine information from Mac OS X metadata.
 *
 * @param {Object} metadata - The metadata object containing 'com.apple.quarantine' property.
 * @returns {Object} An object containing quarantine type, id, source, and UUID.
 */
function extractQuarantineInfo({ 'com.apple.quarantine': quarantineInfo }) {
  // Extract relevant information from the metadata using destructuring
  const [type, id, source, uuid] = quarantineInfo.split(';');

  // Return the extracted information as an object using object literal shorthand
  return {
    type,
    id,
    source,
    uuid,
  };
}

// Example usage:
const macOsMetadata = {
  'com.apple.quarantine': 'q/0281;69525879;Chrome;DB56A116-B245-4A5A-A8CE-690469EE5E44',
};

const parsedMetadata = extractQuarantineInfo(macOsMetadata);
console.log(parsedMetadata);
