// skills/skill.model.js
/**
 * Represents a skill with a name and description.
 */
class Skill {
  #name;
  #description;

  /**
   * Creates a new Skill instance.
   * @param {Object} options - The options for the skill.
   * @param {string} options.name - The name of the skill.
   * @param {string} options.description - The description of the skill.
   */
  constructor({ name, description }) {
    this.#name = name;
    this.#description = description;
  }

  /**
   * Gets the name of the skill.
   * @returns {string} The name of the skill.
   */
  get name() {
    return this.#name;
  }

  /**
   * Gets the description of the skill.
   * @returns {string} The description of the skill.
   */
  get description() {
    return this.#description;
  }

  /**
   * Converts the skill to a JSON object.
   * @returns {Object} The skill as a JSON object.
   */
  toJSON() {
    return {
      name: this.#name,
      description: this.#description,
    };
  }
}

// skills/skill.service.js
import fetch from 'node-fetch';

/**
 * Provides methods for interacting with skills.
 */
class SkillService {
  #baseUrl;

  /**
   * Creates a new SkillService instance.
   * @param {Object} options - The options for the skill service.
   * @param {string} options.baseUrl - The base URL for the skills API.
   */
  constructor({ baseUrl = 'https://example.com' } = {}) {
    this.#baseUrl = baseUrl;
  }

  /**
   * Gets a list of skills from the API.
   * @returns {Promise<Skill[]>} A promise that resolves to a list of skills.
   */
  async getSkills() {
    try {
      const response = await fetch(`${this.#baseUrl}/skills`);
      if (!response.ok) {
        throw new Error(`Failed to fetch skills: ${response.status}`);
      }
      const skillsData = await response.json();
      return skillsData.map((skillData) => new Skill(skillData));
    } catch (error) {
      console.error(error);
      return [];
    }
  }
}

// skills/skill.controller.js
/**
 * Handles requests related to skills.
 */
class SkillController {
  #skillService;

  /**
   * Creates a new SkillController instance.
   * @param {Object} options - The options for the skill controller.
   * @param {SkillService} options.skillService - The skill service to use.
   */
  constructor({ skillService }) {
    this.#skillService = skillService;
  }

  /**
   * Handles a request to get a list of skills.
   * @param {Object} req - The request object.
   * @param {Object} res - The response object.
   */
  async getSkills(req, res) {
    try {
      const skills = await this.#skillService.getSkills();
      res.json(skills.map((skill) => skill.toJSON()));
    } catch (error) {
      console.error(error);
      res.status(500).json({ error: 'Failed to fetch skills' });
    }
  }
}

// index.js
import { SkillController } from './skills/skill.controller.js';
import { SkillService } from './skills/skill.service.js';

const skillService = new SkillService();
const skillController = new SkillController({ skillService });

// example usage
skillController.getSkills({}, (err, skills) => {
  if (err) {
    console.error(err);
  } else {
    console.log(skills);
  }
});
