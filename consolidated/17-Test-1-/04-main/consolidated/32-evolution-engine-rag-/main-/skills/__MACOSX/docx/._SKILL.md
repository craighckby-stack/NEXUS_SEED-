// skills/skill.js
/**
 * Represents a skill with a name and an operating system.
 */
class Skill {
  #validOS = ['Mac OS X', 'Windows', 'Linux'];
  #osRegex = /\b(Mac OS X|Windows|Linux)\b/;

  /**
   * Creates a new Skill instance.
   * @param {string} name - The name of the skill.
   * @param {string} os - The operating system of the skill.
   */
  constructor(name, os) {
    this.name = name;
    this.os = os;
  }

  /**
   * Checks if the given operating system is valid.
   * @param {string} os - The operating system to check.
   * @returns {boolean} True if the operating system is valid, false otherwise.
   */
  static isValidOS(os) {
    return Skill.prototype.#validOS.includes(os);
  }

  /**
   * Parses the given string to extract the operating system.
   * @param {string} osString - The string to parse.
   * @returns {string|null} The extracted operating system or null if not found.
   */
  static parseOS(osString) {
    const match = osString.match(Skill.prototype.#osRegex);
    return match ? match[1] : null;
  }
}

// skills/utils.js
/**
 * Provides utility functions for skills.
 */
class SkillUtils {
  /**
   * Filters skills by valid operating systems.
   * @param {Skill[]} skills - The skills to filter.
   * @returns {Skill[]} The filtered skills.
   */
  filterValidSkills(skills) {
    return skills.filter(skill => Skill.isValidOS(skill.os));
  }
}

// skills/index.js

import { SkillUtils } from './utils.js';

const skills = [

];
