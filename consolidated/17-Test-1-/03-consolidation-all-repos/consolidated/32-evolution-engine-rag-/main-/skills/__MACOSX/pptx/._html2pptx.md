import fs from 'fs/promises';
import path from 'path';

/**
 * Reads a binary file and logs its contents as a string.
 * 
 * @param {string} filePath - The path to the binary file.
 */
async function handleBinaryFile(filePath) {
  try {
    const fileBuffer = await fs.readFile(filePath);
    const fileString = fileBuffer.toString('utf8');
    globalThis.console.log(fileString);
  } catch (error) {
    globalThis.console.error(`Error reading file: ${error}`);
  }
}

// Example usage:
const filePath = path.join(import.meta.url, '..', 'skills/__MACOSX/pptx/._html2pptx.md');
handleBinaryFile(filePath);
