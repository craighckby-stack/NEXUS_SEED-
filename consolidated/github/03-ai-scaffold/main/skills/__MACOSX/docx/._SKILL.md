      await runScript('install', [
        '-D',
        'tailwindcss',
        'postcss',
        'autoprefixer',
      ]);
      await runScript('install', ['clsx', 'tailwind-merge']);
    };
    // Initialize Tailwind
    const initTailwind = async () => {
      await import('npx');
      await runScript('tailwindcss', ['init', '-p']);
      await copyFile('templates/tailwind.config.js', 'tailwind.config.js');
      await copyFile('templates/globals.css', 'src/globals.css');
    };
    // Import in your app
    const importInApp = async () => {
      // React: import './globals.css' in main entry
      // Next.js: import './globals.css' in _app.tsx or layout.tsx
    };
    await installDeps();
    await initTailwind();
    await importInApp();
