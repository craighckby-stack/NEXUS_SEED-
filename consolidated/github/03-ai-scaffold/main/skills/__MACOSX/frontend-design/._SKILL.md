# Frontend Design Skill

A comprehensive skill for transforming UI style requirements into production-ready frontend code with systematic design tokens, accessibility compliance, and creative execution.

## Project Overview
____________________

This skill enables developers to craft visually appealing and accessible frontend interfaces. By following the guidelines outlined in this document, teams can ensure consistency in design and functionality across applications.

______________________

### TypeScript Theme Provider
```typescript
import { ThemeProvider, CSSProperties } from 'styled-components';
import { DesignTokens } from './design-tokens';

interface ThemeContext {
  designTokens: DesignTokens;
  mode: 'light' | 'dark';
}

const ThemeProviderComponent: React.FC = ({ children }) => {
  const [theme, setTheme] = useState<ThemeContext>({
    designTokens,
    mode: 'light',
  });

  const toggleMode = () => {
    setTheme((prevTheme) => ({ ...prevTheme, mode: prevTheme.mode === 'light' ? 'dark' : 'light' }));
  };

  return (
    <ThemeProvider theme={theme}>
      <button onClick={toggleMode}>Toggle Mode</button>
      {children}
    </ThemeProvider>
  );
};
```

### Example Component
```typescript
import { styled } from 'styled-components';

  border: none;
  border-radius: ${(props: any) => props.theme.designTokens.spacing['2']};
  cursor: pointer;
`;

const ExampleComponent: React.FC = () => {
  return <Button>Click me!</Button>;
};
```
