import { readFile } from 'fs/promises';

/**
 * Asynchronously reads the contents of a file.
 * 
 * @param {string} filePath - The path to the file to be read.
 * @returns {Promise<string | null>} A promise that resolves with the file contents as a string, or null if an error occurs.
 */
async function readFileAsync(filePath) {
  try {
    return await readFile(filePath, 'utf8');
  } catch (error) {
    console.error(`Error reading file: ${error}`);
    return null;
  }
}

/**
 * Synchronously reads the contents of a file.
 * 
 * @param {string} filePath - The path to the file to be read.
 * @returns {string | null} The file contents as a string, or null if an error occurs.
 */
function readFileSync(filePath) {
  try {
    return require('fs').readFileSync(filePath, 'utf8');
  } catch (error) {
    console.error(`Error reading file: ${error}`);
    return null;
  }
}
