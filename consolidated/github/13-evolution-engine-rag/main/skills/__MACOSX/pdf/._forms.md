// Define quarantine info object with clear property names
const quarantineInfo = {
  quarantineAttribute: 'q/0281;69525879;Chrome;DB56A116-B245-4A5A-A8CE-690469EE5E44',
  sourceApplication: 'com.apple.quarantine'
};

// Create a function to parse quarantine info with a descriptive name
function extractQuarantineDetails(quarantineString) {
  // Split the quarantine string into parts
  const [quarantineType, ...infoParts] = quarantineString.split(';');
  
  // Extract quarantine details using reduce method
  const quarantineDetails = {
    type: quarantineType.replace('q/', ''),
    ...infoParts.reduce((accumulator, part) => {
      // Split each part into key-value pairs
      const [key, value] = part.includes(':') ? part.split(':') : [part, ''];
      return { ...accumulator, [key]: value };
    }, {})
  };
  
  return quarantineDetails;
}

// Use the function to parse quarantine info
const parsedQuarantineInfo = extractQuarantineDetails(quarantineInfo.quarantineAttribute);

// Log the parsed quarantine info
console.log(parsedQuarantineInfo);
