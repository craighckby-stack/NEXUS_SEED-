// skills/pptx/Skill.js
/**
 * Represents a skill with a name and attributes.
 */
class Skill {
  #name;
  #attributes;

  /**
   * Creates a new Skill instance.
   * @param {string} name - The name of the skill.
   * @param {object} attributes - The attributes of the skill.
   */
  constructor(name, attributes) {
    this.#name = name;
    this.#attributes = attributes;
  }

  /**
   * Returns the name of the skill.
   * @returns {string} The name of the skill.
   */
  get name() {
    return this.#name;
  }

  /**
   * Returns the attributes of the skill.
   * @returns {object} The attributes of the skill.
   */
  get attributes() {
    return this.#attributes;
  }

  /**
   * Returns a string representation of the skill.
   * @returns {string} A string representation of the skill.
   */
  toString() {
    return `Skill: ${this.name}, Attributes: ${JSON.stringify(this.attributes)}`;
  }
}

// skills/pptx/QuarantineAttribute.js
/**
 * Represents a quarantine attribute with a type and value.
 */
class QuarantineAttribute {
  #type;
  #value;

  /**
   * Creates a new QuarantineAttribute instance.
   * @param {string} type - The type of the quarantine attribute.
   * @param {string} value - The value of the quarantine attribute.
   */
  constructor(type, value) {
    this.#type = type;
    this.#value = value;
  }

  /**
   * Returns the type of the quarantine attribute.
   * @returns {string} The type of the quarantine attribute.
   */
  get type() {
    return this.#type;
  }

  /**
   * Returns the value of the quarantine attribute.
   * @returns {string} The value of the quarantine attribute.
   */
  get value() {
    return this.#value;
  }

  /**
   * Returns a string representation of the quarantine attribute.
   * @returns {string} A string representation of the quarantine attribute.
   */
  toString() {
    return `Quarantine Attribute: ${this.type}, Value: ${this.value}`;
  }
}
