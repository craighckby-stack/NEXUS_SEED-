// skills/frontend-design/Skill.js

/**
 * Represents the frontend design skill.
 */
class FrontendDesignSkill {
  #name;
  #description;

  /**
   * Creates a new instance of FrontendDesignSkill.
   * @param {Object} options - The skill options.
   * @param {string} options.name - The name of the skill.
   * @param {string} options.description - The description of the skill.
   */
  constructor({ name = 'Frontend Design', description = 'Design and develop visually appealing and user-friendly interfaces.' } = {}) {
    this.#name = name;
    this.#description = description;
  }

  /**
   * Returns the name of the skill.
   * @returns {string} The name of the skill.
   */
  get name() {
    return this.#name;
  }

  /**
   * Returns the description of the skill.
   * @returns {string} The description of the skill.
   */
  get description() {
    return this.#description;
  }
}

// Example usage:
const frontendDesignSkill = new FrontendDesignSkill();
console.log(frontendDesignSkill.name); // Output: Frontend Design
console.log(frontendDesignSkill.description); // Output: Design and develop visually appealing and user-friendly interfaces.
