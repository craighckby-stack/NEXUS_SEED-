// Define constants for macOS attributes
const MAC_OS_X_ATTRIBUTES = {
  LAST_USED_DATE: 'com.apple.lastuseddate',
  PROVENANCE: 'com.apple.provenance',
  QUARANTINE: 'com.apple.quarantine',
};

// Define a function to parse macOS attributes
function parseMacOSAttributes(attributes) {
  const parsedAttributes = {};

  // Iterate over each attribute
  Object.keys(attributes).forEach((key) => {
    switch (key) {
      case MAC_OS_X_ATTRIBUTES.LAST_USED_DATE:
        parsedAttributes.lastUsedDate = attributes[key];
        break;
      case MAC_OS_X_ATTRIBUTES.PROVENANCE:
        parsedAttributes.provenance = attributes[key];
        break;
      case MAC_OS_X_ATTRIBUTES.QUARANTINE:
        parsedAttributes.quarantine = attributes[key];
        break;
      default:
        break;
    }
  });

  return parsedAttributes;
}

// Example usage:
const attributes = {
  'com.apple.lastuseddate': '2022-01-01',
  'com.apple.provenance': 'Chrome',
  'com.apple.quarantine': '99670C22-663D-4E2D-A26A-8F9C518B770B',
};

const parsedAttributes = parseMacOSAttributes(attributes);
console.log(parsedAttributes);
