/**
 * Skills Markdown Generator
 * @module skills/generator
 */

// Import necessary modules
import { skills } from './skills.js';

/**
 * Generates a markdown string from the provided skills data.
 * @param {Object[]} skillsData - An array of skill objects with name and description properties.
 * @returns {string} The generated markdown string.
 */
function generateMarkdown(skillsData) {
  const markdownLines = ['# Skills\n\n'];

  skillsData.forEach(({ name, description }) => {
    markdownLines.push(`## ${name}\n`);
    markdownLines.push(`${description}\n\n`);
  });

  return markdownLines.join('');
}

// Example usage:
const exampleSkills = [
  { name: 'Programming', description: 'Proficient in JavaScript, Python, and Java' },
  { name: 'Web Development', description: 'Experienced with React, Angular, and Vue' },
];

const generatedMarkdown = generateMarkdown(exampleSkills);
console.log(generatedMarkdown);
