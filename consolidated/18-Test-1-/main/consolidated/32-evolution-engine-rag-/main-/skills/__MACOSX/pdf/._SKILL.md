// skills/skill.js
/**
 * Represents a skill with a name and an operating system.
 */
class Skill {
  /**
   * Creates a new Skill instance.
   * @param {string} name - The name of the skill.
   * @param {string} operatingSystem - The operating system associated with the skill.
   */
  constructor(name, operatingSystem) {
    this.name = name;
    this.operatingSystem = operatingSystem;
  }

  /**
   * Parses a macOS X quarantine attribute.
   * @param {string} attribute - The quarantine attribute to parse.
   * @returns {Object|null} The parsed quarantine attribute, or null if the attribute is invalid.
   */
  static parseMacOSXQuarantineAttribute(attribute) {
    const QUARANTINE_REGEX = /^q\/(\d+);(\d+);(.+);(.+)$/;
    const match = attribute.match(QUARANTINE_REGEX);
    return match ? {
      type: 'quarantine',
      value: match[1],
      timestamp: parseInt(match[2], 10),
      source: match[3],
      identifier: match[4],
    } : null;
  }
}

// Example usage:
const pdfSkill = new Skill('PDF', 'Mac OS X');
const quarantineAttribute = 'q/0281;69525879;Chrome;DB56A116-B245-4A5A-A8CE-690469EE5E44';
const parsedQuarantineAttribute = Skill.parseMacOSXQuarantineAttribute(quarantineAttribute);
console.log(parsedQuarantineAttribute);
