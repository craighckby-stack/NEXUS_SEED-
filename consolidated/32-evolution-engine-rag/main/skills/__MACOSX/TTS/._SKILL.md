/**
 * URL Parser Module
 * 
 * This module provides a function to parse a given URL and extract its components.
 */

/**
 * Represents a parsed URL.
 * 
 * @typedef {Object} ParsedUrl
 * @property {string} protocol - The URL protocol.
 * @property {string} hostname - The URL hostname.
 * @property {string} pathname - The URL pathname.
 * @property {URLSearchParams} searchParams - The URL search parameters.
 */

/**
 * Parses a URL and returns its components.
 * 
 * @param {string} url - The URL to be parsed.
 * @returns {ParsedUrl} The parsed URL components.
 */
function parseUrl(url) {
  const parsedUrl = new URL(url);
  return {
    protocol: parsedUrl.protocol,
    hostname: parsedUrl.hostname,
    pathname: parsedUrl.pathname,
    searchParams: parsedUrl.searchParams,
  };
}

/**
 * Logs the components of a parsed URL.
 * 
 * @param {ParsedUrl} parsedUrl - The parsed URL components.
 */
function logUrlComponents(parsedUrl) {
  console.log(`Protocol: ${parsedUrl.protocol}`);
  console.log(`Hostname: ${parsedUrl.hostname}`);
  console.log(`Path: ${parsedUrl.pathname}`);
  console.log(`Query Parameters: ${parsedUrl.searchParams.toString()}`);
}

/**
 * Example usage of the parseUrl function.
 */
function main() {
  const url = 'https://claude.ai/api/organizations/646bcdf2-2de6-4395-8ef5-efda92eddefa/conversations/9671d65f-ae02-4464-bbd1-3649146c66b7/wiggle/download-file?path=%2Fmnt%2Fuser-data%2Foutputs%2FTTS%2FSKILL.md';
  const parsedUrl = parseUrl(url);
  logUrlComponents(parsedUrl);
}

main();
