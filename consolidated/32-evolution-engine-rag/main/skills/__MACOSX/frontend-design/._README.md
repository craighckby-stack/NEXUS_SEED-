import fs from 'fs/promises';

/**
 * Reads a file safely, handling potential errors.
 * 
 * @param {string} filePath - The path to the file to be read.
 * @returns {Promise<string | null>} The file content as a string, or null if an error occurs.
 */
async function readFileSafely(filePath) {
  try {
    const fileBuffer = await fs.readFile(filePath);
    return fileBuffer.toString();
  } catch (error) {
    if (error.code === 'EISDIR') {
      console.error(`Error: ${filePath} is a directory.`);
    } else if (error.code === 'ENOENT') {
      console.error(`Error: ${filePath} does not exist.`);
    } else {
      console.error(`Error reading file: ${error}`);
    }
    return null;
  }
}

// Example usage:
const filePath = 'path/to/your/file';
readFileSafely(filePath).then((fileContent) => {
  if (fileContent) {
    console.log(fileContent);
  }
});
